//순차 자료구조를 이용해 순차 큐 구현하기

#include "stdio.h"
#include "stdlib.h"
#define Q_SIZE 4

typedef char element; // 큐 원소 (element)의 자료형을 char 로 정의

typedef struct
{
	element queue[Q_SIZE];//1차원 배열 큐 선언
	int front,rear;
}QueueType;

//공백 순차 큐를 생성하는 연산
QueueType *createQueue()
{
	QueueType *Q;
	Q = (QueueType*)malloc(sizeof(QueueType)); //포인터 구조체 Q를 동적할당으로 선언
	Q -> front = -1;// front 초기값 -1로 설정
	Q -> rear = -1; //rear 초기값 -1 로 설정
	return 0;
}
//순차 큐가 공백 상태인지 검사하는 연산
int isEmpty(QueueType *Q)
{
	if(Q -> front == Q -> rear)
	{
		printf("큐 비었음\n");
		return 1;
	}
	else
		return 0;
}

//순차 큐가 포화 상태인지 검사하는 연산
int isFull(QueueType *Q)
{
	if(Q -> rear == Q_SIZE -1)
	{
		printf("큐 가득참\n");
		return 1;
	}
	else
		return 0;
}

//순차 큐의 rear에 원소를 삽입하는 연산
void enQueue(QueueType *Q,element item)
{
	if(isFull(Q))
		return; //포화 상태이면 삽입연산 중단
	else
	{
		Q -> rear++;
		Q -> queue[Q -> rear]=item;
	}
}

//순차 큐의 front 에서 원소를 삭제하는 연산
element deQueue(QueueType *Q)
{
	if(isEmpty(Q))
		return '\0'; // 공백상태이면 삭제 연산 중단
	else
	{
		 Q -> front++;
		 return Q -> queue[Q->front];
	}
}

//순차 큐의 가장 앞에 있는 원소를 검색하는 연산
element peek(QueueType *Q)
{
	if(isEmpty(Q))
		exit(1); // 공백 상태이면 연산 중단
	else
		return Q -> queue[Q -> front +1]; // front 값을 -1 로 초기화 해서 +1 해야 한다
}

//순차 큐의 원소를 출력하는 연산
void printQ(QueueType *Q)
{
	int i;
	printf(" Queue [ ");
	for(i=Q -> front +1;i <= Q -> rear;i++)
		printf("%3c",Q -> queue[i]);
	printf(" ]");
}

void main ()
{
	QueueType *Q1 = createQueue(); // 큐생성
	element data;
	printf("\n****순차 큐 연산*****\n");
	printf("\n삽입 A>>");
	enQueue(Q1,'A');
	printQ(Q1);
	printf("\n삽입 B>>");enQueue(Q1,'B');printQ(Q1);
	printf("\n삽입 C>>");enQueue(Q1,'C');printQ(Q1);
	data = peek(Q1); 
	printf("peek item : %c\n",data);
	printf("\n삭제 >> "); data = deQueue(Q1);printQ(Q1);
	printf("\t삭제 데이터 : %c",data);
	printf("\n삭제 >>");data = deQueue(Q1);printQ(Q1);
	printf("\t삭제 데이터 : %c",data);
	printf("\n삭제 >>");data = deQueue(Q1);printQ(Q1);
	printf("\n삭제 데이터 :%c",data);

	printf("\n 삽입 D>>"); enQueue(Q1,'D');printQ(Q1);
	printf("\n 삽입 E>>"); enQueue(Q1,'E');printQ(Q1);
	getchar();
}
