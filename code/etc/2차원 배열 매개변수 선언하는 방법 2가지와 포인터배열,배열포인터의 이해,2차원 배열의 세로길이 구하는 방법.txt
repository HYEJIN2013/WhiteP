#include<stdio.h>

void ShowArr2Dstyle(int (*arr)[4],int column)//int 형을 자료형으로 가지는 가로가 4인 2차원배열을 가리키는 포인터 arr // 배열요소 전체출력
{
	int i,j;
	for(i=0;i<column;i++)
	{
		for(j=0;j<4;j++)
			printf("%d ",arr[i][j]);
		printf("\n");
	}
	printf("\n");
}

int Sum2DArr(int arr[][4],int column)//int형을 자료형으로 가지는 가로값이 4인 2차원 배열 arr//배열요소의 합 반환
{
	int i,j,sum=0;
	for(i=0;i<column;i++)
		for(j=0;j<4;j++)
			sum += arr[i][j];
	return sum;
}

int main (void)
{
	int arr1[2][4]={1,2,3,4,5,6,7,8};
	int arr2[3][4]={1,1,1,1,3,3,3,3,5,5,5,5};

	ShowArr2Dstyle(arr1,sizeof(arr1)/sizeof(arr1[0]));//sizeof(arr1)/sizeof(arr1[0]) 을 하게 되면 arr1의 세로 길이가 나오게 된다.
	ShowArr2Dstyle(arr2,sizeof(arr2)/sizeof(arr2[0]));

	printf("arr1의 합 : %d\n",Sum2DArr(arr1,sizeof(arr1)/sizeof(arr1[0])));
	printf("arr2의 합:  %d\n",Sum2DArr(arr2,sizeof(arr2)/sizeof(arr2[0])));
	
	//sizeof(arr1)과sizeof(arr2)는 배열의 전체 크기이고 sizeof(arr1[0])과 sizeof(arr2[0])은 배열의 가로 크기이다
	//이는 2차원 배열의 세로길이 계산시 흔히 사용되는 방법이니 기억해놔야한다 ! 
	
	return 0;
}


//매개변수 선언시 int(*arr)[4]{배열 포인터} 와 int arr[][4]는 같다.!!!!!!

//배열 포인터 - 포인터는 포인터 인데 배열을 가리키는 포인터
//포인터 배열 - 배열은 배열이지만 포인터로 이루어진 배열!
