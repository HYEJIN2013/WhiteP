#include<stdio.h>//2차원 배열의 선언과 동시에 초기화 하는방법을 예시로 들어놨다,1차원 배열과 똑같이 빈공간은 0으로 초기화 됨이 보인다

int main (void)
{
	int i,j;

	/* 2차원 배열 초기화의 예시 1. */

	int arr1[3][3]={
		{1,2,3},
		{4,5,6},
		{7,8,9}
	};

	/* 2차원 배열 초기화의 예시 2. */

	int arr2[3][3]={
		{1},
		{4,5},
		{7,8,9}
	};

	/* 2차원 배열 초기화의 예시 3. */

	int arr3[3][3]={1,2,3,4,5,6,7};

	for(i=0;i<3;i++)
	{
		for(j=0;j<3;j++)
			printf("%d ",arr1[i][j]);
		printf("\n");
	}
	printf("\n");
	for(i=0;i<3;i++)
	{
		for(j=0;j<3;j++)
			printf("%d ",arr2[i][j]);
		printf("\n");
	}
	printf("\n");
	for(i=0;i<3;i++)
	{
		for(j=0;j<3;j++)
			printf("%d ",arr3[i][j]);
		printf("\n");
	}
	printf("\n");
	return 0;
}
